Standard showcase skeleton:
```iron
import raylib

func main() -> Int32 {
    -- Window init
    Window.init(Int32(800), Int32(600), "raylib_showcase")
    Window.set_target_fps(Int32(60))

    -- Audio init
    Audio.init()

    -- Load assets
    val blip = Sound.load("tests/assets/bounce.wav")
    val model = Model.load("tests/assets/models/cube.obj")
    val shader = Shader.load("", "tests/assets/shaders/grayscale.fs")

    -- Procedural 2D texture (NO new asset)
    val checker = Image.checked(Int32(64), Int32(64), Int32(8), Int32(8), RED, BLUE)
    val tex = checker.to_texture()
    checker.unload()

    -- Camera3D setup
    var cam = Camera3D(
        Vector3(Float32(5.0), Float32(5.0), Float32(5.0)),
        Vector3(Float32(0.0), Float32(0.0), Float32(0.0)),
        Vector3(Float32(0.0), Float32(1.0), Float32(0.0)),
        Float32(45.0),
        Int32(0)  -- CameraProjection.PERSPECTIVE.ordinal workaround (73-01 tracked)
    )

    -- MATH: raymath evaluated once before loop
    val forward = Vector3(Float32(0), Float32(0), Float32(1)).normalize()

    -- FILE: evaluated once before loop
    val toml_exists = Files.exists("iron.toml")

    -- Main loop
    while not Window.should_close() {
        -- INPUT
        if Keyboard.is_pressed(KeyboardKey.SPACE) {
            Sound.play(blip)
        }

        -- COLL: check collision between two hardcoded rects
        val r1 = Rectangle.of(Float32(10), Float32(10), Float32(50), Float32(50))
        val r2 = Rectangle.of(Float32(30), Float32(30), Float32(50), Float32(50))
        val hit = Shapes.check_collision_rec_rec(r1, r2)

        -- DRAW2D
        Draw.begin()
        Draw.clear(RAYWHITE)
        Draw.rectangle(Int32(10), Int32(10), Int32(100), Int32(50), RED)
        Texture.draw(tex, Int32(200), Int32(200), WHITE)

        -- DRAW3D + MODEL
        Draw.begin_mode_3d(cam)
        Model.draw(model, Vector3(Float32(0), Float32(0), Float32(0)), Float32(1.0), WHITE)
        Draw.grid(Int32(10), Float32(1.0))
        Draw.end_mode_3d()

        -- TEXT
        Draw.text("raylib_showcase — SPACE for sound, ESC to exit", Int32(10), Int32(570), Int32(20), DARKGRAY)

        -- SHADER (nested shader mode toggles)
        Draw.begin_shader_mode(shader)
        Texture.draw(tex, Int32(300), Int32(300), WHITE)
        Draw.end_shader_mode()

        Draw.end()
    }

    -- Cleanup
    Texture.unload(tex)
    Sound.unload(blip)
    Shader.unload(shader)
    Model.unload(model)
    Audio.close()
    Window.close()
    return Int32(0)
}
```
