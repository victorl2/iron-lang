ann->layout_hint = IRON_LAYOUT_HINT_NONE;
ann->is_unordered = false;

/* Parse optional attributes: [T, layout: soa] or [T, unordered] or [T, layout: aos, unordered] */
while (iron_match(p, IRON_TOK_COMMA)) {
    Iron_Token *attr = iron_current(p);
    if (iron_check(p, IRON_TOK_IDENTIFIER)) {
        if (strcmp(attr->value, "layout") == 0) {
            iron_advance(p);  /* consume "layout" */
            iron_expect(p, IRON_TOK_COLON);
            Iron_Token *val = iron_current(p);
            if (iron_check(p, IRON_TOK_IDENTIFIER)) {
                if (strcmp(val->value, "soa") == 0) {
                    ann->layout_hint = IRON_LAYOUT_HINT_SOA;
                    iron_advance(p);
                } else if (strcmp(val->value, "aos") == 0) {
                    ann->layout_hint = IRON_LAYOUT_HINT_AOS;
                    iron_advance(p);
                } else {
                    iron_diag_emit(p->diags, p->arena, IRON_DIAG_ERROR,
                        IRON_ERR_UNEXPECTED_TOKEN,
                        iron_token_span(p, val),
                        "expected 'soa' or 'aos' after 'layout:'", NULL);
                    iron_advance(p);
                }
            }
        } else if (strcmp(attr->value, "unordered") == 0) {
            ann->is_unordered = true;
            iron_advance(p);
        } else {
            iron_diag_emit(p->diags, p->arena, IRON_DIAG_ERROR,
                IRON_ERR_UNEXPECTED_TOKEN,
                iron_token_span(p, attr),
                "expected 'layout' or 'unordered' in array type attribute", NULL);
            iron_advance(p);
        }
    }
}

**Part A: Large variant indirection in tagged unions**

In `emit_type_decls()` in emit_c.c, where the data union is generated (around line 4830-4848), add variant size analysis:

1. Before emitting the union members, compute the "size" of each alive implementor. Approximate size by counting fields and their types:
   - Int/Bool: 8 bytes (int64_t/bool)
   - Float: 8 bytes (double)
   - String: 16 bytes (Iron_String = pointer + length)
   - Array field: 24 bytes (pointer + count + cap)
   - Object field: recursive, but estimate 8 bytes (pointer) for simplicity
   - Unknown: 8 bytes

   The size estimation function:
   ```c
   static int estimate_type_size(Iron_ObjectDecl *od) {
       int total = 0;
       for (int i = 0; i < od->field_count; i++) {
           Iron_Field *f = (Iron_Field *)od->fields[i];
           if (f->type_ann) {
               Iron_TypeAnnotation *ta = (Iron_TypeAnnotation *)f->type_ann;
               if (ta->is_array) total += 24;
               else if (ta->is_func) total += 16; /* Iron_Closure */
               else if (strcmp(ta->name, "String") == 0) total += 16;
               else total += 8;
           } else {
               total += 8;
           }
       }
       return total > 0 ? total : 8;
   }
   ```

2. Find the smallest and largest alive implementor sizes. Check thresholds:
   - `largest_size > 2 * smallest_size AND largest_size > 64`
   - If true: the large variant(s) get pointer indirection

3. For large variants, emit `Iron_<Type> *<TypeName>;` (pointer) instead of `Iron_<Type> <TypeName>;` (inline) in the data union.

4. Update the wrapping constructors for large variants (around line 4849-4867):
   ```c
   /* For large variant (pointer indirection): */
   static inline Iron_Shape Iron_Shape_from_BigImpl(Iron_BigImpl val) {
       Iron_Shape result;
       result.tag = Iron_Shape_TAG_BigImpl;
       result.data.BigImpl = (Iron_BigImpl*)malloc(sizeof(Iron_BigImpl));
       *result.data.BigImpl = val;
       return result;
   }
   ```

5. For GET_FIELD on an interface-typed value where the matched variant is a large (pointer) variant, emit `->field` instead of `.field`. This affects the dispatch switch in emit_instr GET_FIELD handler (around line 1434-1470).

   Add a tracking set: `struct { char *key; bool value; } *indirect_variants;` in EmitCtx, keyed by `"iface:type"`. Populated during the union emission. Consulted during GET_FIELD dispatch.

6. In the free function generation (or when the interface variable goes out of scope), free the heap-allocated large variant. Add `free()` call for pointer variants in the SplitList free function and in any scope-exit cleanup.
