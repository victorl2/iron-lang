**Step 1: Guard existing plain-enum path.** In the enum emission loop (line 3130), add a check: `if (ed->has_payloads)` to branch to the new ADT emission path. When `!ed->has_payloads`, the existing code runs unchanged.

**Step 2: ADT emission path.** When `ed->has_payloads == true`, write to `ctx->struct_bodies` (NOT `ctx->enum_defs`). Emit the following C layout (using the mangled enum name):

```c
/* Tag enum */
typedef enum {
    Shape_TAG_Circle = 0,
    Shape_TAG_Rect = 1,
} Shape_Tag;

/* Per-variant payload structs (only for variants with payload_count > 0) */
typedef struct { double _0; } Shape_Circle_data;
typedef struct { double _0; double _1; } Shape_Rect_data;

/* Union of payloads */
typedef union {
    char _dummy;          /* C99 requires at least one member */
    Shape_Circle_data Circle;
    Shape_Rect_data   Rect;
} Shape_data_t;

/* The ADT struct */
typedef struct {
    Shape_Tag    tag;
    Shape_data_t data;
} Shape;
```

Concrete implementation:
- Tag enum: iterate `ed->variants`, emit `{mangled}_TAG_{ev->name} = {j}` for each variant. Use `iron_strbuf_appendf(&ctx->struct_bodies, ...)`.
- Per-variant payload structs: for each variant where `ev->payload_count > 0`, iterate `ev->payload_types[k]` (0..payload_count-1). Use `emit_type_to_c(ev->payload_types[k], ctx)` to get the C type string. Field names are `_0`, `_1`, etc.
- Union: always include `char _dummy;` first. Then for each variant with payloads, add `{mangled}_{ev->name}_data {ev->name};`.
- Outer struct: tag field uses `{mangled}_Tag tag;`, data field uses `{mangled}_data_t data;`.

**Step 3: Also add forward declaration.** Write `typedef struct {mangled} {mangled};` into `ctx->forward_decls` for ADT enums (so other structs can reference them as pointers before the full definition appears).

**Step 4: Extend CONSTRUCT emission** (line 1764). Add a branch for `instr->construct.type->kind == IRON_TYPE_ENUM && ed->has_payloads`:
- The CONSTRUCT for an ADT enum has field_vals in order: [tag_int, payload_0, payload_1, ...].
- Emit as designated initializer: `Shape _vN = { .tag = Shape_TAG_Circle, .data.Circle = { val0, val1 } };`
- Alternatively, emit as: `Shape _vN = { .tag = (Shape_Tag)tag_val, .data = { .Circle = { val0, val1 } } };`
- To determine the variant name: look up the tag integer value from field_vals[0] (it is a CONST_INT instruction) and find `ed->variants[tag_value]->name`.
- Payload values are field_vals[1..field_count-1].

**Step 5: Ensure `emit_type_to_c` handles `IRON_TYPE_ENUM` with `has_payloads`.** Currently it probably returns the mangled name directly (which is correct — the typedef makes `Shape` a valid C type). Verify this works and fix if needed.

**Part C: Add lowering cases in hir_lower.c `lower_expr_hir()`.**

Before the `default:` case (line 1084), add:

```c
case IRON_NODE_ENUM_CONSTRUCT: {
    Iron_EnumConstruct *ec = (Iron_EnumConstruct *)node;
    Iron_Type *ty = ec->resolved_type;
    /* Find variant index */
    int variant_idx = -1;
    if (ty && ty->kind == IRON_TYPE_ENUM && ty->enu.decl) {
        Iron_EnumDecl *ed = ty->enu.decl;
        for (int i = 0; i < ed->variant_count; i++) {
            Iron_EnumVariant *ev = (Iron_EnumVariant *)ed->variants[i];
            if (strcmp(ev->name, ec->variant_name) == 0) {
                variant_idx = i;
                break;
            }
        }
    }
    /* Lower args */
    IronHIR_Expr **args = NULL;
    for (int i = 0; i < ec->arg_count; i++) {
        IronHIR_Expr *a = lower_expr_hir(ctx, ec->args[i]);
        arrput(args, a);
    }
    int ac = (int)arrlen(args);
    return iron_hir_expr_enum_construct(mod, ty, ec->enum_name, ec->variant_name,
                                         variant_idx, args, ac, span);
}

case IRON_NODE_PATTERN: {
    Iron_Pattern *pat = (Iron_Pattern *)node;
    /* Find variant index from resolved type context (use parent match scrutinee type) */
    int variant_idx = -1;
    /* Pattern does not carry resolved_type directly; variant_index is looked up
       by iterating the enum decl. We need the enum type from the scrutinee.
       For now, store -1 and let hir_to_lir.c resolve it from the match scrutinee type. */
    /* Lower nested patterns recursively */
    IronHIR_Expr **nested = NULL;
    for (int i = 0; i < pat->binding_count; i++) {
        if (pat->nested_patterns && pat->nested_patterns[i]) {
            IronHIR_Expr *np = lower_expr_hir(ctx, pat->nested_patterns[i]);
            arrput(nested, np);
        } else {
            arrput(nested, NULL);
        }
    }
    /* Copy binding names */
    const char **names = NULL;
    for (int i = 0; i < pat->binding_count; i++) {
        arrput(names, pat->binding_names ? pat->binding_names[i] : NULL);
    }
    return iron_hir_expr_pattern(mod, pat->enum_name, pat->variant_name,
                                  variant_idx, names, nested, pat->binding_count, span);
}
```
