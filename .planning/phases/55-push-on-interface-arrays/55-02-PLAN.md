```c
} else if (strcmp(coll_method, "pop") == 0) {
    /* Phase 55: .pop() on split collection.
     * Consult _order[_total_count - 1] to find (tag, sub_index), read the
     * element from the matching per-type sub-array, wrap in the interface
     * tagged union, decrement counts. Returns the popped element. */
    const char *pop_type = emit_type_to_c(instr->type, ctx);  /* Iron_<Iface> */
    emit_indent(sb, ind);
    iron_strbuf_appendf(sb, "%s ", pop_type);
    emit_val(sb, instr->id);
    iron_strbuf_appendf(sb, ";\n");
    emit_indent(sb, ind);
    iron_strbuf_appendf(sb, "{\n");
    /* Compute last-index once */
    emit_indent(sb, ind + 1);
    iron_strbuf_appendf(sb, "int64_t _sp_pop_i = ");
    emit_val(sb, self_arg);
    iron_strbuf_appendf(sb, "._total_count - 1;\n");
    /* Tag switch dispatches to per-type read + decrement */
    emit_indent(sb, ind + 1);
    iron_strbuf_appendf(sb, "switch (");
    emit_val(sb, self_arg);
    iron_strbuf_appendf(sb, "._order[_sp_pop_i].tag) {\n");
    for (int ji = 0; ji < sp_entry->impl_count; ji++) {
        Iron_IfaceImpl *impl = &sp_entry->impls[ji];
        if (!impl->is_alive) continue;
        /* Lowercase type name for field access */
        char lower_name[256];
        {
            size_t nl2 = strlen(impl->type_name);
            if (nl2 >= sizeof(lower_name)) nl2 = sizeof(lower_name) - 1;
            for (size_t ci3 = 0; ci3 < nl2; ci3++)
                lower_name[ci3] = (char)((impl->type_name[ci3] >= 'A' &&
                                           impl->type_name[ci3] <= 'Z')
                    ? impl->type_name[ci3] + 32
                    : impl->type_name[ci3]);
            lower_name[nl2] = '\0';
        }
        emit_indent(sb, ind + 2);
        iron_strbuf_appendf(sb, "case %d: ", ji);
        emit_val(sb, instr->id);
        iron_strbuf_appendf(sb, " = %s_from_%s(", sp_iface, impl->type_name);
        emit_val(sb, self_arg);
        iron_strbuf_appendf(sb, ".%s_items[", lower_name);
        emit_val(sb, self_arg);
        iron_strbuf_appendf(sb, "._order[_sp_pop_i].idx]); ");
        emit_val(sb, self_arg);
        iron_strbuf_appendf(sb, ".%s_count--; break;\n", lower_name);
    }
    emit_indent(sb, ind + 2);
    iron_strbuf_appendf(sb, "default: memset(&");
    emit_val(sb, instr->id);
    iron_strbuf_appendf(sb, ", 0, sizeof(");
    emit_val(sb, instr->id);
    iron_strbuf_appendf(sb, ")); break;\n");
    emit_indent(sb, ind + 1);
    iron_strbuf_appendf(sb, "}\n");
    /* Decrement order and total counts */
    emit_indent(sb, ind + 1);
    emit_val(sb, self_arg);
    iron_strbuf_appendf(sb, "._order_count--;\n");
    emit_indent(sb, ind + 1);
    emit_val(sb, self_arg);
    iron_strbuf_appendf(sb, "._total_count--;\n");
    emit_indent(sb, ind);
    iron_strbuf_appendf(sb, "}\n");
    break;
}
```

**Scoping constraint — SoA implementors:** the per-type read `<lower>_items[...]` assumes AoS storage. If any alive impl uses SoA storage (check via `shgeti(ctx->soa_types, "<sp_iface>:<type_name>")` — see emit_split.c:391-394 for the exact key format), the read path differs: per-field arrays instead of one _items array. For this plan, if any alive impl is SoA, skip the full implementation and emit a fallback that zero-initializes the result via `memset(&<id>, 0, sizeof(<id>));` plus a C comment `/* SoA pop not yet supported */`. Document as known limitation in commit message. The common case (AoS) must work.

**Note on `pop_type`:** `instr->type` should be the interface tagged-union type (`Iron_<Iface>`). If Iron's HIR types pop as `Option<T>` or some wrapper, adapt accordingly. If `emit_type_to_c` returns something surprising, surface and adjust.

**Note on empty-collection pop:** the `default:` case handles edge cases (no matching tag). For true safety, insert an `if (_sp_pop_i >= 0)` guard, but the simple version above is sufficient for the test cases (pop is only called when the collection is known non-empty). Document this as a known limitation if needed.

**Note on the order of decrements:** the sub-array `<lower>_count` is decremented inside the case (so the corresponding sub-array logically shrinks), and `_order_count` / `_total_count` are decremented after the switch. This is the correct order: per-type count tracks sub-array occupancy, total count tracks the whole collection.
