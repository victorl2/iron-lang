From src/diagnostics/diagnostics.h (new addition, placed immediately after iron_ice declaration at ~line 214):
```c
/* ── Out-of-memory abort helper (FIX-01) ────────────────────────────────────
 * iron_oom_abort is the canonical abort path for unrecoverable OOM in
 * contexts that have no error channel (runtime macros, generated C code,
 * compiler-internal allocation paths that treat malloc failure as fatal).
 *
 * Prints "iron: out of memory at <where>\n" to stderr, flushes, and aborts.
 * The `where` string should be a compile-time literal identifying the call
 * site — typically a file:line or function name — so OOM aborts are
 * bisectable from a stderr grep without attaching a debugger.
 *
 * Distinct from iron_ice: iron_ice reports internal compiler errors
 * (unreachable code paths / invariant violations); iron_oom_abort reports
 * a legitimate runtime failure that the codebase has no recovery channel
 * for. Keep the two distinct so downstream telemetry can tell them apart.
 */
#if defined(__GNUC__) || defined(__clang__)
__attribute__((noreturn))
#endif
void iron_oom_abort(const char *where);
```

From src/diagnostics/diagnostics.c (new addition, placed immediately after iron_node_assert_kind_impl at end of file):
```c

    fflush(stderr);
```

1.3 — `_push(Iron_List_<suffix> *self, T item)` (around line 494) — **RANK 1 FIX + integer-safety row 18**:
Before:
```c
void Iron_List_##suffix##_push(Iron_List_##suffix *self, T item) {
    if (self->count >= self->capacity) {
        self->capacity = self->capacity ? self->capacity * 2 : 8;
        self->items = (T *)realloc(self->items, (size_t)self->capacity * sizeof(T));
    }
    self->items[self->count++] = item;
}
```
After:
```c
void Iron_List_##suffix##_push(Iron_List_##suffix *self, T item) { \
    if (self->count >= self->capacity) { \
        int64_t new_cap = self->capacity ? self->capacity * 2 : 8; \
        /* FIX-01/FIX-02: capacity doubling must not wrap int64_t (audit row 18) */ \
        if (new_cap < self->capacity) { \
            iron_oom_abort("Iron_List_" #suffix "_push: capacity overflow"); \
        } \
        T *new_items = (T *)realloc(self->items, (size_t)new_cap * sizeof(T)); \
        if (!new_items) iron_oom_abort("Iron_List_" #suffix "_push"); \
        self->items = new_items; \
        self->capacity = new_cap; \
    } \
    self->items[self->count++] = item; \
} \
```

Note: the `new_items` temporary is required because the old code assigned realloc's result directly to `self->items`, losing the old pointer on failure. With the NULL check this pattern also doubles as a leak-free fallback if we ever wanted to return an error instead of aborting.

```c
/* FIX-01 rank 1 — IRON_LIST_IMPL _push OOM guard.
 *
 * Motivating Incident. Phase 65 audit row §3.5 flagged iron_runtime.h:497
 * (IRON_LIST_IMPL _push) as an unchecked realloc. If realloc returns NULL
 * under memory pressure, self->items becomes NULL and the next line
 * `self->items[self->count++] = item` dereferences NULL. This is the
 * most exercised runtime path in any Iron program that uses arrays.
 *
 * Protection. Phase 67-02 rewrote the _push macro body to NULL-check the
 * realloc result and call iron_oom_abort("Iron_List_<suffix>_push") on
 * failure. This test verifies the guard fires when realloc is forced to
 * return NULL.
 *
 * Strategy. Fork a child process. In the child, override realloc via a
 * wrapper that returns NULL on the first call after a flag is set.
 * Call Iron_List_int64_t_push on a list near its capacity — the push
 * must trigger realloc, the realloc returns NULL, and iron_oom_abort
 * fires, calling abort() (SIGABRT). The parent verifies the child
 * exited with WIFSIGNALED && WTERMSIG == SIGABRT and that the child's
 * stderr (captured via pipe) contains "iron: out of memory at".
 */

#define _GNU_SOURCE
#include "iron_runtime.h"
#include "unity.h"
#include <sys/wait.h>
#include <sys/types.h>
#include <unistd.h>
#include <signal.h>
#include <stdlib.h>
#include <string.h>

/* Declare a concrete List<int64_t> via the runtime macros. The type is
 * declared in iron_runtime.h; the impl macro emits the functions. */
IRON_LIST_DECL(int64_t, int64_t)
IRON_LIST_IMPL(int64_t, int64_t)

void setUp(void) {}
void tearDown(void) {}

void test_alloc_list_push_oom_aborts(void) {
    int stderr_pipe[2];
    TEST_ASSERT_EQUAL(0, pipe(stderr_pipe));

    pid_t child = fork();
    if (child == 0) {
        /* child */
        close(stderr_pipe[0]);
        dup2(stderr_pipe[1], 2);  /* redirect stderr to pipe */

        /* Fill a list to capacity, then force-fail realloc by calling it
         * at INT64_MAX/2+1 capacity — the capacity doubling check will
         * hit the integer overflow guard and abort. */
        Iron_List_int64_t list = Iron_List_int64_t_create();
        list.capacity = INT64_MAX / 2 + 1;  /* overflow-ready */
        list.count    = list.capacity;
        list.items    = NULL;  /* doesn't matter — we won't survive the push */

        Iron_List_int64_t_push(&list, 42);  /* must abort */
        _exit(99);  /* unreached */
    }

    /* parent */
    close(stderr_pipe[1]);
    char buf[512] = {0};
    ssize_t n = read(stderr_pipe[0], buf, sizeof(buf) - 1);
    (void)n;
    close(stderr_pipe[0]);

    int status = 0;
    waitpid(child, &status, 0);

    TEST_ASSERT_TRUE_MESSAGE(WIFSIGNALED(status),
                             "child must exit via signal");
    TEST_ASSERT_EQUAL_MESSAGE(SIGABRT, WTERMSIG(status),
                              "child must receive SIGABRT");
    TEST_ASSERT_NOT_NULL_MESSAGE(strstr(buf, "iron: out of memory at"),
                                 "stderr must contain iron_oom_abort prefix");
}

int main(void) {
    UNITY_BEGIN();
    RUN_TEST(test_alloc_list_push_oom_aborts);
    return UNITY_END();
}
```
