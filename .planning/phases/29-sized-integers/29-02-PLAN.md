Key patterns for benchmark rewrite:
- `val x: Int32 = 42` -- Int literal implicitly narrows to Int32
- `Int32(someInt)` -- explicit cast from Int to Int32
- `fill(50, Int32(0))` -- creates [Int32] stack array (50 <= 256)
- `arr[i]` on [Int32] returns Int32
- `someInt32 < someInt` -- comparison auto-widens (works)
- `someInt32 + someInt32` -- arithmetic stays Int32 (works)
- `someInt32 + someInt` -- mixed arithmetic is a TYPE ERROR (not allowed)

Critical constraint: Array indexing uses Int index. If `node` is Int32 and used as
array index `parent[node]`, this works because C implicitly converts int32_t to int
for array subscript. However, watch for the `while parent[node] != node` pattern --
both sides must be Int32 for the comparison to work without widening issues.

The connected_components functions that take Int index parameters (x, y) should
become Int32 since the indices are node IDs that fit in 32 bits.

```iron
func find(parent: [Int32], x: Int32) -> [Int32] {
    var p = parent
    var node = x
    while p[node] != node {
        p[node] = p[p[node]]
        node = p[node]
    }

}
```

**unite():**
```iron
func unite(parent: [Int32], rnk: [Int32], x: Int32, y: Int32) -> [Int32] {
    var p = parent
    var r = rnk
    var rx = find_root(p, x)
    var ry = find_root(p, y)
    if rx != ry {
        if r[rx] < r[ry] {
            val t = rx
            rx = ry
            ry = t
        }
        p[ry] = rx
        if r[rx] == r[ry] {
            r[rx] = r[rx] + Int32(1)
        }
    }
    return p
}
```
- All params become Int32/[Int32]

**count_components():**
```iron
func count_components(edges_a: [Int32], edges_b: [Int32], num_edges: Int, n: Int) -> Int {
    var parent = fill(50, Int32(0))
    var rnk = fill(50, Int32(0))
    for i in range(n) {
        parent[i] = Int32(i)
    }

    var e = 0
    while e < num_edges {
        val x: Int32 = edges_a[e]
        val y: Int32 = edges_b[e]
        val rx = find_root(parent, x)
        val ry = find_root(parent, y)
        if rx != ry {
            var real_rx = rx
            var real_ry = ry
            if rnk[real_rx] < rnk[real_ry] {
                val t = real_rx
                real_rx = real_ry
                real_ry = t
            }
            parent[real_ry] = real_rx
            if rnk[real_rx] == rnk[real_ry] {
                rnk[real_rx] = rnk[real_rx] + Int32(1)
            }
        }
        e = e + 1
    }

    var count = 0
    for i in range(n) {
        if find_root(parent, Int32(i)) == Int32(i) {
            count = count + 1
        }
    }
    return count
}
```
- `edges_a: [Int]` -> `edges_a: [Int32]`, same for edges_b
- Keep `num_edges: Int` and `n: Int` since they are loop bounds (Int indexing)
- `fill(50, 0)` -> `fill(50, Int32(0))` to produce [Int32]
- `parent[i] = i` -> `parent[i] = Int32(i)` since `i` is Int from range() but parent expects Int32

```iron
import time

-- Int32 array sum: measures 32-bit array throughput
-- Sums all elements of an Int32 array using 32-bit arithmetic

func array_sum(arr: [Int32], n: Int) -> Int32 {
    var total: Int32 = Int32(0)
    var i = 0
    while i < n {
        total = total + arr[i]
        i = i + 1
    }
    return total
}

func main() {
    val n = 200

    -- Build array: arr[i] = i % 100
    var arr = fill(n, Int32(0))
    var i = 0
    while i < n {
        arr[i] = Int32(i % 100)
        i = i + 1
    }

    -- Test cases
    println("Test 1: {array_sum(arr, n)} (expected 9900)")

    -- Partial sum: first 50 elements (0+1+...+49 = 1225)
    println("Test 2: {array_sum(arr, 50)} (expected 1225)")

    -- Single element
    var single = fill(1, Int32(42))
    println("Test 3: {array_sum(single, 1)} (expected 42)")

    -- Empty (0 elements)
    println("Test 4: {array_sum(arr, 0)} (expected 0)")

    val iterations = 10000000
    val start = Time.now_ms()
    var result: Int32 = Int32(0)
    for it in range(iterations) {
        result = array_sum(arr, n)
    }
    val elapsed = Time.now_ms() - start

    println("")
    println("=== Benchmark: Int32 Array Sum ===")
    println("Array size: {n}")
    println("Iterations: {iterations}")
    println("Total time: {elapsed} ms")
    println("Result: {result}")
}
```

```c
#include <stdio.h>
#include <stdint.h>
#include <time.h>

static int32_t array_sum(const int32_t *arr, int n) {
    int32_t total = 0;
    for (int i = 0; i < n; i++) {
        total += arr[i];
    }
    return total;
}

int main(void) {
    const int n = 200;
    int32_t arr[200];
    for (int i = 0; i < n; i++) {
        arr[i] = (int32_t)(i % 100);
    }

    printf("Test 1: %d (expected 9900)\n", array_sum(arr, n));
    printf("Test 2: %d (expected 1225)\n", array_sum(arr, 50));

    int32_t single[1] = {42};
    printf("Test 3: %d (expected 42)\n", array_sum(single, 1));
    printf("Test 4: %d (expected 0)\n", array_sum(arr, 0));

    const int iterations = 10000000;
    struct timespec ts_start, ts_end;
    clock_gettime(CLOCK_MONOTONIC, &ts_start);
    int32_t result = 0;
    for (int it = 0; it < iterations; it++) {
        result = array_sum(arr, n);
    }
    clock_gettime(CLOCK_MONOTONIC, &ts_end);
    long elapsed = (ts_end.tv_sec - ts_start.tv_sec) * 1000 +
                   (ts_end.tv_nsec - ts_start.tv_nsec) / 1000000;

    printf("\n");
    printf("=== Benchmark: Int32 Array Sum ===\n");
    printf("Array size: %d\n", n);
    printf("Iterations: %d\n", iterations);
    printf("Total time: %ld ms\n", elapsed);
    printf("Result: %d\n", result);
    return 0;
}
```
