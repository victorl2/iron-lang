```c
/* Implicit coercion: Int <-> Int32 for assignments.
 * - Int literal -> Int32: narrowing (always safe for literals)
 * - Int32 -> Int: widening (always safe, no data loss)
 */
static bool types_assignable(const Iron_Type *decl_t, const Iron_Type *init_t) {
    if (!decl_t || !init_t) return true;
    if (iron_type_equals(decl_t, init_t)) return true;
    /* Int -> Int32: implicit narrowing for integer values */
    if (decl_t->kind == IRON_TYPE_INT32 && init_t->kind == IRON_TYPE_INT) return true;
    /* Int32 -> Int: implicit widening (always safe) */
    if (decl_t->kind == IRON_TYPE_INT && init_t->kind == IRON_TYPE_INT32) return true;
    return false;
}
```

Then replace `!iron_type_equals(X, Y)` with `!types_assignable(X, Y)` at these 6 assignment-check sites:

    return x
}

func main() {
    -- val with Int32 annotation and integer literal

    println("{x}")

    -- Int32() cast from Int
    val n: Int = 100
    val y: Int32 = Int32(n)

    -- Int32 arithmetic (same-width)
    val z: Int32 = x + y
    println("{z}")

    -- Function with Int32 params and return
    val sum: Int32 = add_int32(Int32(10), Int32(20))
    println("{sum}")

    -- Widening: Int32 -> Int (implicit)
    val w: Int = z
    println("{w}")

    -- Function returning widened Int32 -> Int
    val wide = widen_to_int(Int32(7))
    println("{wide}")

    -- Comparison: Int32 < Int (auto-widening)
    val limit: Int = 200
    if z < limit {
        println("less")
    } else {
        println("not less")
    }

    -- Comparison: Int == Int32 (auto-widening)
    val check: Int = 42
    if check == x {
        println("equal")
    } else {
        println("not equal")
    }
}
