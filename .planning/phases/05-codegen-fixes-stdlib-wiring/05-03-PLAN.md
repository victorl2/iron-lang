```c
case IRON_NODE_METHOD_CALL: {
    Iron_MethodCallExpr *mc = (Iron_MethodCallExpr *)node;

    /* AUTO-STATIC CHECK: If the receiver is a type name (IRON_SYM_TYPE),
     * emit a direct C function call without self pointer.
     * Convention: Iron_<lowercase_type>_<method>(args...)
     * This matches the stdlib C API naming: Iron_math_sin, Iron_io_read_file, etc. */
    if (mc->object->kind == IRON_NODE_IDENT) {
        Iron_Ident *obj_id = (Iron_Ident *)mc->object;
        if (obj_id->resolved_sym &&
            obj_id->resolved_sym->sym_kind == IRON_SYM_TYPE) {
            /* Auto-static: emit Iron_<lower(type)>_<method>(args...) */
            const char *tname = obj_id->name;  /* e.g. "Math" */

            /* Build lowercase version of type name */
            size_t tlen = strlen(tname);
            char *lower = (char *)iron_arena_alloc(ctx->arena, tlen + 1);
            for (size_t k = 0; k < tlen; k++) {
                lower[k] = (char)tolower((unsigned char)tname[k]);
            }
            lower[tlen] = '\0';

            /* Emit: Iron_<lower>_<method>(args...) */
            iron_strbuf_appendf(sb, "Iron_%s_%s(", lower, mc->method);
            for (int i = 0; i < mc->arg_count; i++) {
                if (i > 0) iron_strbuf_appendf(sb, ", ");
                emit_expr(sb, mc->args[i], ctx);
            }
            iron_strbuf_appendf(sb, ")");
            break;
        }
    }

    /* ... existing instance method dispatch code below (unchanged) ... */
```

This code:
- Checks if the receiver ident's resolved_sym is IRON_SYM_TYPE
- Constructs the C function name as `Iron_` + lowercase(typename) + `_` + methodname
- Emits a function call WITHOUT self pointer
- Math.sin(1.0) -> Iron_math_sin(1.0) -- correct!
- IO.read_file("test.txt") -> Iron_io_read_file(...) -- correct!
