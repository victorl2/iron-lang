/* TEX-01 */
struct Iron_Image Iron_image_load(Iron_String path) {
    const char *cpath = iron_string_cstr(&path);
    Image src = LoadImage(cpath ? cpath : "");
    struct Iron_Image out;
    memcpy(&out, &src, sizeof(struct Iron_Image));

void Iron_image_unload(struct Iron_Image img) {
    Image src;
    memcpy(&src, &img, sizeof(Image));
    UnloadImage(src);

bool Iron_image_is_valid(struct Iron_Image img) {
    Image src;
    memcpy(&src, &img, sizeof(Image));
    return (bool)(IsImageValid(src) != 0);
}

/* LoadImageAnim — frames out-param ignored here; user passes the slot as Iron Int (0 OK).
 * NOTE: Current Iron ABI has no out-ref; `frames` slot is technically unused in the shim.
 * Phase 67+ follow-up may add an accessor or tuple-return variant. */
struct Iron_Image Iron_image_load_anim(Iron_String path, int64_t frames) {
    (void)frames;  /* silence -Wunused-parameter */
    const char *cpath = iron_string_cstr(&path);
    int raylib_frames = 0;
    Image src = LoadImageAnim(cpath ? cpath : "", &raylib_frames);
    struct Iron_Image out;
    memcpy(&out, &src, sizeof(struct Iron_Image));
    return out;
}

struct Iron_Image Iron_image_from_texture(struct Iron_Texture tex) {
    Texture t;
    memcpy(&t, &tex, sizeof(Texture));
    Image src = LoadImageFromTexture(t);
    struct Iron_Image out;
    memcpy(&out, &src, sizeof(struct Iron_Image));
    return out;
}
