Modify the `IRON_NODE_METHOD_CALL` case in `lower_expr()` (src/ir/lower_exprs.c, around line 353-377) to detect auto-static dispatch and emit a direct function call instead of a method call with receiver.

The current code (line 353-377) always does:
1. `lower_expr(ctx, mc->object)` -- tries to resolve `Math` as a variable, which fails with "unresolved identifier"
2. Passes the object value as the first argument (receiver)

The fix: before lowering the object, check if the receiver is an identifier that resolves to `IRON_SYM_TYPE`. If so, this is an auto-static call -- emit a direct function call to `<lowercase_type>_<method>` with NO receiver argument.

Replace the `case IRON_NODE_METHOD_CALL:` block with:

```c
case IRON_NODE_METHOD_CALL: {
    Iron_MethodCallExpr *mc = (Iron_MethodCallExpr *)node;

    /* AUTO-STATIC CHECK: If the receiver is a type name (IRON_SYM_TYPE),
     * emit a direct C function call without self pointer.
     * Convention: <lowercase_type>_<method>(args...)
     * The emit_c.c mangle_func_name() will add the Iron_ prefix at emission.
     * This matches the stdlib C API: Iron_math_sin, Iron_io_read_file, etc. */
    if (mc->object->kind == IRON_NODE_IDENT) {
        Iron_Ident *obj_id = (Iron_Ident *)mc->object;
        if (obj_id->resolved_sym &&
            obj_id->resolved_sym->sym_kind == IRON_SYM_TYPE) {
            /* Build the function name: lowercase(TypeName)_methodName */
            const char *tname = obj_id->name;
            size_t tlen = strlen(tname);
            size_t mlen = strlen(mc->method);
            size_t total = tlen + 1 + mlen + 1;  /* type_method\0 */
            char *func_name = (char *)iron_arena_alloc(ctx->ir_arena, total,
                                                        _Alignof(char));
            /* Lowercase the type name */
            for (size_t k = 0; k < tlen; k++) {
                func_name[k] = (char)tolower((unsigned char)tname[k]);
            }
            func_name[tlen] = '_';
            memcpy(func_name + tlen + 1, mc->method, mlen + 1);

            /* Emit func_ref for the auto-static function name */
            IronIR_ValueId func_ptr = iron_ir_func_ref(fn, ctx->current_block,
                                                        func_name, NULL, span)->id;

            /* Lower arguments (NO receiver for auto-static) */
            IronIR_ValueId *args = NULL;
            for (int i = 0; i < mc->arg_count; i++) {
                IronIR_ValueId v = lower_expr(ctx, mc->args[i]);
                arrput(args, v);
            }

            IronIR_ValueId result = iron_ir_call(fn, ctx->current_block,
                                                  NULL, func_ptr,
                                                  args, mc->arg_count,
                                                  mc->resolved_type, span)->id;
            arrfree(args);
            return result;
        }
    }

    /* Instance method call: receiver is an object instance */
    IronIR_ValueId obj_val = lower_expr(ctx, mc->object);

    /* Build args: receiver first, then explicit args */
    IronIR_ValueId *args = NULL;
    arrput(args, obj_val);
    for (int i = 0; i < mc->arg_count; i++) {
        IronIR_ValueId v = lower_expr(ctx, mc->args[i]);
        arrput(args, v);
    }
    int total_args = 1 + mc->arg_count;

    /* Emit method name as func_ref */
    IronIR_ValueId func_ptr = iron_ir_func_ref(fn, ctx->current_block,
                                                mc->method, NULL, span)->id;
    IronIR_ValueId result = iron_ir_call(fn, ctx->current_block,
                                          NULL, func_ptr,
                                          args, total_args,
                                          mc->resolved_type, span)->id;
    arrfree(args);
    return result;
}
```

Add support for top-level `val` declarations (including comptime constants) in the IR lowerer. Currently, `lower_func_bodies()` only processes `IRON_NODE_FUNC_DECL` nodes, completely ignoring top-level `IRON_NODE_VAL_DECL` and `IRON_NODE_VAR_DECL`.

The comptime_basic.iron test has:
```iron
val GREETING = comptime "hello comptime"
func main() {
    println(GREETING)
}
```

After comptime evaluation, the AST has `GREETING` as a `IRON_NODE_VAL_DECL` with `init` pointing to a `IRON_NODE_STRING_LIT("hello comptime")`. When `main()` references `GREETING`, the identifier lookup fails because `GREETING` was never registered in the lowering context.

The fix has two parts:

**Part A: Register top-level constants before function body lowering**

In `src/ir/lower.c`, add a new pass between `lower_module_decls()` and `lower_func_bodies()` inside `iron_ir_lower()`. This pass iterates program->decls looking for `IRON_NODE_VAL_DECL` (and `IRON_NODE_VAR_DECL` if needed), and emits them as global constants.

Add this function before `lower_func_bodies()`:

```c
/* Lower top-level val/var declarations as global constants.
 * After comptime evaluation, top-level val GREETING = comptime "..."
 * becomes val GREETING = STRING_LIT "...". We lower the init expression
 * in the context of the main function's entry block, then store the
 * ValueId in val_binding_map so any function body can reference it.
 *
 * Strategy: Emit each global constant's initializer at the start of
 * Iron_main's entry block. Record the resulting ValueId in a
 * global_constants map on the context so that IDENT lookups resolve
 * across function boundaries. */
static void lower_global_constants(IronIR_LowerCtx *ctx) {
    /* Find the main function in the module */
    IronIR_Func *main_fn = NULL;
    for (int i = 0; i < ctx->module->func_count; i++) {
        if (strcmp(ctx->module->funcs[i]->name, "main") == 0) {
            main_fn = ctx->module->funcs[i];
            break;
        }
    }
    if (!main_fn) return;  /* No main function — skip global constants */

    /* We need to emit the constant initializers before the main function body.
     * Since lower_func_body hasn't been called yet, main_fn has no blocks.
     * We'll collect the top-level val names and init expressions, then
     * during lower_func_body(main), we'll emit them at the start. */

    /* Alternative approach: use a per-context hash map for global constants.
     * Record (name -> init_node) pairs. During IDENT resolution, check
     * this map and lower the init expression lazily in the current function. */

    for (int i = 0; i < ctx->program->decl_count; i++) {
        Iron_Node *decl = ctx->program->decls[i];
        const char *var_name = NULL;
        Iron_Node  *init_expr = NULL;

        if (decl->kind == IRON_NODE_VAL_DECL) {
            Iron_ValDecl *vd = (Iron_ValDecl *)decl;
            var_name  = vd->name;
            init_expr = vd->init;
        } else if (decl->kind == IRON_NODE_VAR_DECL) {
            Iron_VarDecl *vd = (Iron_VarDecl *)decl;
            var_name  = vd->name;
            init_expr = vd->init;
        }

        if (!var_name || !init_expr) continue;

        /* Store in global_constants_map: name -> AST init node */
        shput(ctx->global_constants_map, var_name, init_expr);
    }
}
```

/* Check global_constants_map (top-level val/comptime declarations) */
{
