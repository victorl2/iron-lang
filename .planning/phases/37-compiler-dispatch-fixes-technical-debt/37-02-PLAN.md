```c
#ifndef IRON_IMPORT_DETECT_H
#define IRON_IMPORT_DETECT_H

#include <stdbool.h>
#include "../util/arena.h"

/**
 * Returns true if `source` contains a real `import <module_name>` statement.
 *
 * Uses the Iron lexer — immune to false positives from `import` inside
 * comments or string literals. Scans for IRON_TOK_IMPORT immediately followed
 * by IRON_TOK_IDENTIFIER whose value equals module_name.
 *
 * @param source      Null-terminated source text (not modified, not freed)
 * @param filename    Source filename for diagnostic spans (may be NULL → "<import-detect>")
 * @param module_name Module name to search for (e.g. "math", "io", "raylib")
 * @param arena       Caller-owned arena used for token allocation (contents discarded after call)
 * @return            true if a real `import <module_name>` token sequence is present
 */
bool iron_detect_import(const char *source, const char *filename,
                        const char *module_name, Iron_Arena *arena);

#endif /* IRON_IMPORT_DETECT_H */
```

**src/cli/iron_import_detect.c:**
```c
#include "iron_import_detect.h"
#include "../lexer/lexer.h"
#include "../diagnostics/diagnostics.h"
#include <string.h>

bool iron_detect_import(const char *source, const char *filename,
                        const char *module_name, Iron_Arena *arena) {
    if (!source || !module_name) return false;
    const char *fname = filename ? filename : "<import-detect>";

    /* Create a throwaway diagnostic list — lex errors are ignored for detection */
    Iron_DiagList diags = iron_diaglist_create();

    Iron_Lexer lexer = iron_lexer_create(source, fname, arena, &diags);
    Iron_Token *tokens = iron_lex_all(&lexer);

    bool found = false;
    if (tokens) {
        ptrdiff_t count = arrlen(tokens);
        for (ptrdiff_t i = 0; i < count - 1; i++) {
            if (tokens[i].kind == IRON_TOK_IMPORT &&
                tokens[i + 1].kind == IRON_TOK_IDENTIFIER &&
                tokens[i + 1].value != NULL &&
                strcmp(tokens[i + 1].value, module_name) == 0) {
                found = true;
                break;
            }
        }
        arrfree(tokens);
    }

    iron_diaglist_free(&diags);
    return found;
}
```

Note: The arena used for token allocation is the caller's arena — its memory is released by the caller. The helper does not create or free arenas; it borrows the caller's. This matches how build.c already manages a local arena for the compilation pipeline.

Check the exact Iron_Token struct field name for identifier text — it may be `.value`, `.text`, or `.lexeme`. Read lexer.h Iron_Token definition to confirm before writing the strcmp call.
