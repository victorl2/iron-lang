```c
/* Phase 56 Plan 02: Push arg-type compatibility check.
 * Returns true if `arg_type` can be pushed onto an array of `elem_type`.
 * Rules:
 *   - Exact primitive kind match (Int == Int, not Int == Float)
 *   - Same object decl (Circle == Circle, Square != Circle)
 *   - Concrete implementing interface (Circle implements Shape -> can push Circle onto [Shape])
 *   - Interface == interface (if the user annotated [Shape] and pushes a Shape-typed value)
 */
static bool push_type_compatible(Iron_Type *elem_type, Iron_Type *arg_type) {
    if (!elem_type || !arg_type) return true;  /* permissive on error types */
    if (elem_type->kind == IRON_TYPE_ERROR || arg_type->kind == IRON_TYPE_ERROR) return true;

    /* Direct kind match for primitives */
    if (elem_type->kind == IRON_TYPE_INT || elem_type->kind == IRON_TYPE_FLOAT ||
        elem_type->kind == IRON_TYPE_BOOL || elem_type->kind == IRON_TYPE_STRING) {
        return elem_type->kind == arg_type->kind;
    }

    /* Object == Object: same decl required */
    if (elem_type->kind == IRON_TYPE_OBJECT && arg_type->kind == IRON_TYPE_OBJECT) {
        return elem_type->object.decl == arg_type->object.decl;
    }

    /* Interface elem accepting an object arg: arg's decl must implement the interface */
    if (elem_type->kind == IRON_TYPE_INTERFACE && arg_type->kind == IRON_TYPE_OBJECT &&
        elem_type->interface.decl && arg_type->object.decl) {
        const char *iface_name = elem_type->interface.decl->name;
        Iron_ObjectDecl *od = arg_type->object.decl;
        for (int i = 0; i < od->implements_count; i++) {
            if (od->implements_names[i] && strcmp(od->implements_names[i], iface_name) == 0) {
                return true;
            }
        }
        return false;
    }

    /* Interface elem accepting an interface arg: same decl pointer */
    if (elem_type->kind == IRON_TYPE_INTERFACE && arg_type->kind == IRON_TYPE_INTERFACE) {
        return elem_type->interface.decl == arg_type->interface.decl;
    }

    /* Conservative fallback: reject unknown combinations to surface as diagnostics */
    return false;
}
```
