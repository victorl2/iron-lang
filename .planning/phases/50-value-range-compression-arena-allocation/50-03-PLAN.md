-- value_range_compress.iron: Value range compression (Phase 50)
-- Tests VRC-01: fields proven to hold small ranges stored as narrower types
-- Tests VRC-02: widening reads and narrowing writes produce correct results

interface Particle {
  func describe() -> Int
}

object Dot {
  kind: Int
  x: Int
  y: Int
  func describe() -> Int {
    return self.kind * 1000 + self.x * 100 + self.y
  }
}

object Pixel {
  kind: Int
  r: Int
  g: Int
  func describe() -> Int {
    return self.kind * 1000 + self.r * 100 + self.g
  }
}

func main() {
  -- kind fields only hold 1 or 2, x/y hold 10-50, r/g hold 0-255
  -- All should be compressible to uint8_t
  let particles: [Particle] = [
    Dot(1, 10, 20),
    Dot(1, 30, 40),
    Pixel(2, 100, 200),
    Pixel(2, 50, 150)
  ]

  var total = 0
  for p in particles {
    total = total + p.describe()
  }
  println(total)

  -- Verify individual values are preserved through compression/widening
  let dots: [Particle] = [Dot(1, 50, 99)]
  for d in dots {
    println(d.describe())
  }
}
```

- total = 2020 + 4040 + 12200 + 7150 = 25410

-- arena_split_collection.iron: Arena allocation for split collections (Phase 50)
-- Tests ARENA-01: per-type sub-arrays use arena-tracked allocation with geometric growth
-- Tests ARENA-02: arena free releases all sub-arrays in one operation

interface Animal {
  func sound() -> Int
}

object Dog {
  volume: Int
  func sound() -> Int { return self.volume * 10 }
}

object Cat {
  pitch: Int
  func sound() -> Int { return self.pitch * 5 }
}

object Bird {
  freq: Int
  func sound() -> Int { return self.freq * 2 }
}

func main() {
  -- Create a larger collection to trigger geometric growth (>8 elements initial cap)
  let animals: [Animal] = [
    Dog(3), Cat(7), Bird(100),
    Dog(5), Cat(2), Bird(200),
    Dog(1), Cat(9), Bird(50),
    Dog(8), Cat(4), Bird(75),
    Dog(6), Cat(1)
  ]

  var total = 0
  for a in animals {
    total = total + a.sound()
  }
  println(total)

  -- Verify count
  println(14)

  -- Small collection (within initial capacity 8)
  let pets: [Animal] = [Dog(2), Cat(3)]
  var petsum = 0
  for p in pets {
    petsum = petsum + p.sound()
  }
  println(petsum)
}
```
