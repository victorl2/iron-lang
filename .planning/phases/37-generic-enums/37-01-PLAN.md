**parser.c — Fix type annotation generic args to support nested generics:**
The current `iron_parse_type_annotation` at line 248 calls `iron_parse_generic_params` for generic args. This only parses simple identifiers (`Int`, `String`), which means `Result[Option[Int], String]` would fail because `Option[Int]` is not a simple identifier.

Replace the generic args parsing (lines 248-254) with a new approach that calls `iron_parse_type_annotation` recursively for each arg:
```c
/* generic args: Type[T, U] where T, U are full type annotations */
if (iron_check(p, IRON_TOK_LBRACKET)) {
    iron_advance(p);  /* consume '[' */
    iron_skip_newlines(p);
    ann->generic_args = NULL;
    ann->generic_arg_count = 0;
    while (!iron_check(p, IRON_TOK_RBRACKET) && !iron_check(p, IRON_TOK_EOF)) {
        Iron_Node *arg = iron_parse_type_annotation(p);
        arrput(ann->generic_args, arg);
        ann->generic_arg_count++;
        iron_skip_newlines(p);
        if (!iron_match(p, IRON_TOK_COMMA)) break;
        iron_skip_newlines(p);
    }
    iron_expect(p, IRON_TOK_RBRACKET);
} else {
    ann->generic_args      = NULL;
    ann->generic_arg_count = 0;
}
```
This allows `Result[Option[Int], String]` to parse correctly because each arg is parsed as a full type annotation (which itself can have generic args).

**Important:** The existing call to `iron_parse_generic_params` for enum DECLARATIONS (`enum Option[T]`) is correct because T, E are simple identifiers. Only type annotation USE sites need full type annotation parsing.

**typecheck.c — Add substitute_generic_type helper function:**
Add a static helper function (before resolve_type_annotation) that recursively substitutes generic param types:
```c
/* Recursively substitute IRON_TYPE_GENERIC_PARAM with concrete types.
 * Maps ed->generic_params[i].name -> type_args[i]. */
static Iron_Type *substitute_generic_type(TypeCtx *ctx, Iron_Type *t,
                                           Iron_EnumDecl *ed,
                                           Iron_Type **type_args,
                                           int type_arg_count) {
    if (!t) return t;
    switch (t->kind) {
        case IRON_TYPE_GENERIC_PARAM: {
            /* Find the param index by name */
            for (int i = 0; i < type_arg_count && i < ed->generic_param_count; i++) {
                Iron_Ident *param = (Iron_Ident *)ed->generic_params[i];
                if (param && t->generic_param.name &&
                    strcmp(param->name, t->generic_param.name) == 0) {
                    return type_args[i];
                }
            }
            return t; /* unmatched param, return as-is */
        }
        case IRON_TYPE_NULLABLE: {
            Iron_Type *inner = substitute_generic_type(ctx, t->nullable.inner,
                                                        ed, type_args, type_arg_count);
            if (inner == t->nullable.inner) return t; /* no change */
            return iron_type_make_nullable(ctx->arena, inner);
        }
        case IRON_TYPE_ARRAY: {
            Iron_Type *elem = substitute_generic_type(ctx, t->array.elem,
                                                       ed, type_args, type_arg_count);
            if (elem == t->array.elem) return t;
            return iron_type_make_array(ctx->arena, elem, t->array.size);
        }
        case IRON_TYPE_ENUM: {
            /* Nested generic enum: substitute each type_arg recursively */
            if (t->enu.type_arg_count == 0) return t;
            bool changed = false;
            Iron_Type **new_args = iron_arena_alloc(ctx->arena,
                sizeof(Iron_Type *) * (size_t)t->enu.type_arg_count,
                _Alignof(Iron_Type *));
            for (int i = 0; i < t->enu.type_arg_count; i++) {
                new_args[i] = substitute_generic_type(ctx, t->enu.type_args[i],
                                                       ed, type_args, type_arg_count);
                if (new_args[i] != t->enu.type_args[i]) changed = true;
            }
            if (!changed) return t;
            /* Rebuild the monomorphized type with substituted args */
            /* (this case handles deeply nested generics) */
            /* For now, return t — nested generic substitution is deferred */
            return t;
        }
        default:
            return t; /* primitive types, etc. — no substitution needed */
    }
}
```

    iron_strbuf_free(&sb);

    Iron_Type *mono = iron_arena_alloc(ctx->arena, sizeof(Iron_Type),
        _Alignof(Iron_Type));
    memset(mono, 0, sizeof(*mono));
    mono->kind = IRON_TYPE_ENUM;
    mono->enu.decl = ed;
    mono->enu.type_args = inferred_args;
    mono->enu.type_arg_count = ed->generic_param_count;
    mono->enu.mangled_name = mangled;

    /* Substitute variant_payload_types */
    Iron_Type ***vpt = iron_arena_alloc(ctx->arena,
        sizeof(Iron_Type **) * (size_t)ed->variant_count,
        _Alignof(Iron_Type **));
    memset(vpt, 0, sizeof(Iron_Type **) * (size_t)ed->variant_count);
    for (int vj = 0; vj < ed->variant_count; vj++) {
        Iron_EnumVariant *vev = (Iron_EnumVariant *)ed->variants[vj];
        if (vev->payload_count == 0) { vpt[vj] = NULL; continue; }
        Iron_Type **row = iron_arena_alloc(ctx->arena,
            sizeof(Iron_Type *) * (size_t)vev->payload_count,
            _Alignof(Iron_Type *));

        Iron_Scope *saved2 = ctx->global_scope;
        Iron_Scope *gen2 = iron_scope_create(ctx->arena, ctx->global_scope);
        for (int gi = 0; gi < ed->generic_param_count; gi++) {
            Iron_Ident *param = (Iron_Ident *)ed->generic_params[gi];
            if (param) {
                Iron_Type *gpt = iron_type_make_generic_param(ctx->arena, param->name, NULL);
                iron_scope_define(gen2, param->name, gpt);
            }
        }
        ctx->global_scope = gen2;
        for (int kk = 0; kk < vev->payload_count; kk++) {
            Iron_Type *pt = resolve_type_annotation(ctx, vev->payload_type_anns[kk]);
            row[kk] = substitute_generic_type(ctx, pt, ed, inferred_args, ed->generic_param_count);
        }
        ctx->global_scope = saved2;
        vpt[vj] = row;
    }
    mono->enu.variant_payload_types = vpt;

    ec->resolved_type = mono;
    result = mono;
    break;
}
