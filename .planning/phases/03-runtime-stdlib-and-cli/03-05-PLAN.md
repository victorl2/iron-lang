    Header declares, .c implements:
    ```c
    #include <math.h>

    /* Constants */
    #define IRON_PI  3.14159265358979323846
    #define IRON_TAU 6.28318530717958647692
    #define IRON_E   2.71828182845904523536

    /* Trig — thin wrappers returning double */
    double Iron_math_sin(double x);    // return sin(x);
    double Iron_math_cos(double x);    // return cos(x);
    double Iron_math_tan(double x);    // return tan(x);
    double Iron_math_sqrt(double x);   // return sqrt(x);
    double Iron_math_pow(double base, double exp);  // return pow(base, exp);
    double Iron_math_floor(double x);  // return floor(x);
    double Iron_math_ceil(double x);   // return ceil(x);
    double Iron_math_round(double x);  // return round(x);

    /* Lerp */
    double Iron_math_lerp(double a, double b, double t);  // return a + (b - a) * t;

    /* RNG — xorshift64 */
    typedef struct { uint64_t state; } Iron_RNG;
    Iron_RNG Iron_rng_create(uint64_t seed);
    uint64_t Iron_rng_next(Iron_RNG *rng);  // xorshift64 step
    int64_t Iron_rng_next_int(Iron_RNG *rng, int64_t min, int64_t max);
    double Iron_rng_next_float(Iron_RNG *rng);  // 0.0..1.0

    /* Global convenience (uses thread-local RNG seeded from time) */
    double Iron_math_random(void);   // 0.0..1.0
    int64_t Iron_math_random_int(int64_t min, int64_t max);
    ```

    xorshift64 implementation:
    ```c
    uint64_t Iron_rng_next(Iron_RNG *rng) {
        uint64_t x = rng->state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        rng->state = x;
        return x;
    }
    ```

    Global RNG: `static __thread Iron_RNG s_global_rng;` initialized on first use from `clock_gettime`. Use `__thread` for thread-safety.

    **iron_io.h + iron_io.c** — io module:
    ```c
    /* Result types for error returns */
    typedef struct { Iron_String v0; Iron_Error v1; } Iron_Result_String_Error;
    typedef struct { bool v0; Iron_Error v1; } Iron_Result_bool_Error;

    /* File I/O */
    Iron_Result_String_Error Iron_io_read_file(Iron_String path);

    Iron_Result_String_Error Iron_io_read_bytes(Iron_String path);

    /* File system */
    bool Iron_io_file_exists(Iron_String path);
    Iron_Error Iron_io_create_dir(Iron_String path);
    Iron_Error Iron_io_delete_file(Iron_String path);
    /* list_files returns a comma-separated Iron_String for now (v1 simplification) */
    Iron_Result_String_Error Iron_io_list_files(Iron_String dir_path);
    ```

    Implementation uses: `fopen/fread/fwrite/fclose` for file I/O, `stat()` for file_exists, `mkdir()` for create_dir, `opendir/readdir/closedir` for list_files, `remove()` for delete_file.

    read_file:
    1. `const char *p = iron_string_cstr(&path);`
    2. `FILE *f = fopen(p, "rb");` — if NULL, return `{empty_string, iron_error_new(1, "file not found")}`
    3. fseek to end, ftell for size, fseek back
    4. malloc buffer, fread, fclose
    5. Create Iron_String from buffer, free buffer
    6. Return `{string, iron_error_none()}`

    **iron_time.h + iron_time.c** — time module:
    ```c
    #include <time.h>

    /* Monotonic time */
    double Iron_time_now(void);      // seconds since epoch (double)
    int64_t Iron_time_now_ms(void);  // milliseconds monotonic
    void Iron_time_sleep(int64_t ms); // sleep milliseconds

    /* Timer */
    typedef struct { int64_t start_ms; } Iron_Timer;
    Iron_Timer Iron_timer_create(void);
    int64_t Iron_timer_since(const Iron_Timer *t);  // ms elapsed
    void Iron_timer_reset(Iron_Timer *t);
    ```

    Implementation uses `clock_gettime(CLOCK_MONOTONIC, &ts)` on macOS/Linux. Sleep uses `nanosleep()`.

    ```c
    int64_t Iron_time_now_ms(void) {
        struct timespec ts;
        clock_gettime(CLOCK_MONOTONIC, &ts);
        return (int64_t)ts.tv_sec * 1000 + (int64_t)ts.tv_nsec / 1000000;
    }

    void Iron_time_sleep(int64_t ms) {
        struct timespec ts = { .tv_sec = ms / 1000, .tv_nsec = (ms % 1000) * 1000000 };
        nanosleep(&ts, NULL);
    }
    ```

    **iron_log.h + iron_log.c** — log module:

    ```c
    typedef enum { IRON_LOG_DEBUG, IRON_LOG_INFO, IRON_LOG_WARN, IRON_LOG_ERROR } Iron_LogLevel;

    void Iron_log_set_level(Iron_LogLevel level);
    void Iron_log_debug(Iron_String msg);
    void Iron_log_info(Iron_String msg);
    void Iron_log_warn(Iron_String msg);
    void Iron_log_error(Iron_String msg);
    ```

    Implementation:
    - `static Iron_LogLevel s_log_level = IRON_LOG_INFO;` (default)
    - Each function checks level, gets current time (via time module or directly via localtime_r), formats `[YYYY-MM-DD HH:MM:SS] [LEVEL] message\n`, writes to stderr
    - Use `time()` + `localtime_r()` for wall-clock timestamps in log output
    - Use `isatty(STDERR_FILENO)` to gate ANSI color: DEBUG=gray, INFO=green, WARN=yellow, ERROR=red
