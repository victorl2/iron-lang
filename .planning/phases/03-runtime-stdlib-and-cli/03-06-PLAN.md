    **src/cli/main.c** — CLI entry point:
    ```c
    #include <stdio.h>
    #include <stdlib.h>
    #include <string.h>
    #include "cli/build.h"
    #include "cli/check.h"

    static void print_usage(void) {
        fprintf(stderr, "Usage: iron <command> [options] <file>\n\n");
        fprintf(stderr, "Commands:\n");
        fprintf(stderr, "  build   Compile .iron file to native binary\n");
        fprintf(stderr, "  run     Compile and execute .iron file\n");
        fprintf(stderr, "  check   Type-check without compiling\n");
        fprintf(stderr, "  fmt     Format Iron source code\n");
        fprintf(stderr, "  test    Discover and run Iron tests\n");
        fprintf(stderr, "\nOptions:\n");
        fprintf(stderr, "  --verbose       Show generated C code\n");
        fprintf(stderr, "  --debug-build   Keep .iron-build/ directory\n");
    }

    int main(int argc, char **argv) {
        if (argc < 2) { print_usage(); return 1; }

        const char *cmd = argv[1];
        // Parse global flags
        bool verbose = false;
        bool debug_build = false;
        const char *source_file = NULL;
        const char **run_args = NULL;
        int run_arg_count = 0;

        for (int i = 2; i < argc; i++) {
            if (strcmp(argv[i], "--verbose") == 0) { verbose = true; }
            else if (strcmp(argv[i], "--debug-build") == 0) { debug_build = true; }
            else if (strcmp(argv[i], "--") == 0) {
                // Everything after -- is passed to the program (iron run)
                run_args = (const char **)&argv[i + 1];
                run_arg_count = argc - i - 1;
                break;
            }
            else if (!source_file) { source_file = argv[i]; }
        }

        if (strcmp(cmd, "build") == 0) {
            if (!source_file) { fprintf(stderr, "iron build: missing source file\n"); return 1; }
            IronBuildOpts opts = { .verbose = verbose, .debug_build = debug_build };
            return iron_build(source_file, NULL, opts);
        }
        if (strcmp(cmd, "run") == 0) {
            if (!source_file) { fprintf(stderr, "iron run: missing source file\n"); return 1; }
            IronBuildOpts opts = { .verbose = verbose, .debug_build = debug_build, .run_after = true, .run_args = run_args, .run_arg_count = run_arg_count };
            return iron_build(source_file, NULL, opts);
        }
        if (strcmp(cmd, "check") == 0) {
            if (!source_file) { fprintf(stderr, "iron check: missing source file\n"); return 1; }
            return iron_check(source_file, verbose);
        }
        if (strcmp(cmd, "fmt") == 0) {
            fprintf(stderr, "iron fmt: not yet implemented\n"); return 1; // Plan 07
        }
        if (strcmp(cmd, "test") == 0) {
            fprintf(stderr, "iron test: not yet implemented\n"); return 1; // Plan 07
        }
        fprintf(stderr, "iron: unknown command '%s'\n", cmd);
        print_usage();
        return 1;
    }
    ```

    **src/cli/build.c** — Full build pipeline:
    1. Read source file into string (`fopen/fread/fclose`)
    2. Create arena and diag list
    3. Run full pipeline: `iron_lex()` -> `iron_parse()` -> `iron_analyze()` -> `iron_codegen()`
    4. Check diagnostics after each step. If errors, print them using Rust-style formatting and return 1.
    5. If `--verbose`: print generated C to stdout
    6. Determine output binary name: strip `.iron` from source_path (e.g., `hello.iron` -> `hello`)
    7. If output_path provided, use that instead
    8. Write generated C to temp file: use `/tmp/iron_XXXXXX.c` via `mkstemp()` (or `.iron-build/` if `--debug-build`)
    9. Build clang command: `clang -std=c11 -O2 -o <binary> <generated.c> -I<src_dir> -L<lib_dir> -liron_runtime -liron_stdlib -lm -lpthread`
       Actually, for v1 simplicity: compile the runtime source files directly alongside the generated C instead of linking a library. This avoids the need for a pre-built library.
       Command: `clang -std=c11 -O2 -o <binary> <generated.c> src/runtime/iron_string.c src/runtime/iron_rc.c src/runtime/iron_builtins.c src/runtime/iron_threads.c src/runtime/iron_collections.c src/stdlib/iron_math.c src/stdlib/iron_io.c src/stdlib/iron_time.c src/stdlib/iron_log.c -Isrc -Isrc/vendor -lm -lpthread`
       Use `posix_spawn()` or `fork()/execvp()` to invoke clang. Capture exit code.
    10. If clang fails: print error, return 1
    11. If not `--debug-build`: remove temp .c file
    12. If `opts.run_after` (iron run): execute the binary via `execvp` or `fork+exec`, passing run_args. Return the binary's exit code.
    13. Print: `Built: <binary_path>` on success (if not running)

    **Diagnostic formatting** — for Rust-style error output:
    The diagnostics module from Phase 1 already has error printing with source snippets, arrows, and E-codes. Use the existing `iron_diag_print` or equivalent function. For each diagnostic in the diag list, call the print function.

    Color support: The diagnostics module already uses `isatty(STDERR_FILENO)` to gate ANSI colors (established in Phase 1). This should work out of the box.
