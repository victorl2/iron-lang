    ```yaml
    name: CI

    on:
      push:
        branches: [main]
      pull_request:
        branches: [main]

    jobs:
      build-and-test:
        strategy:
          matrix:
            os: [ubuntu-latest, macos-latest]
        runs-on: ${{ matrix.os }}

        steps:
          - uses: actions/checkout@v4

          - name: Install dependencies (Ubuntu)
            if: matrix.os == 'ubuntu-latest'
            run: |
              sudo apt-get update
              sudo apt-get install -y cmake ninja-build clang

          - name: Configure (Debug with ASan/UBSan)
            run: |
              cmake -S . -B build \
                -DCMAKE_BUILD_TYPE=Debug \
                -DCMAKE_C_COMPILER=clang \
                -G Ninja

          - name: Build
            run: cmake --build build

          - name: Run unit tests
            run: ctest --test-dir build --output-on-failure -j4

          - name: Run integration tests
            run: |
              chmod +x tests/integration/run_integration.sh
              tests/integration/run_integration.sh build/iron

          - name: Verify no sanitizer leaks
            run: |
              # Run tests again with ASAN_OPTIONS for stricter checking
              ASAN_OPTIONS=detect_leaks=1:abort_on_error=1 \
              UBSAN_OPTIONS=halt_on_error=1:print_stacktrace=1 \
              ctest --test-dir build --output-on-failure -j4
    ```

    - Debug build in CMakeLists.txt already adds `-fsanitize=address,undefined`
    - macOS runner has clang by default (Apple Clang)

    - Ninja generator for faster builds
    - Two test runs: one normal, one with strict sanitizer options

    **tests/integration/run_integration.sh** — Integration test runner:
    ```bash
    #!/bin/bash
    set -e

    IRON_BIN="${1:-./build/iron}"
    TEST_DIR="$(dirname "$0")"
    PASS=0
    FAIL=0
    TOTAL=0

    echo "=== Iron Integration Tests ==="
    echo "Using: $IRON_BIN"
    echo ""

    for test_file in "$TEST_DIR"/*.iron; do
        test_name=$(basename "$test_file" .iron)
        expected_file="$TEST_DIR/${test_name}.expected"
        TOTAL=$((TOTAL + 1))

        if [ ! -f "$expected_file" ]; then
            echo "[SKIP] $test_name (no .expected file)"
            continue
        fi

        echo -n "[RUN ] $test_name ... "

        # Build the test
        output_bin="/tmp/iron_test_${test_name}"
        if ! "$IRON_BIN" build "$test_file" 2>/tmp/iron_build_err_${test_name}; then
            echo "[FAIL] (build failed)"
            cat /tmp/iron_build_err_${test_name}
            FAIL=$((FAIL + 1))
            continue
        fi

        # Run and capture output
        actual=$("$output_bin" 2>&1) || true

        # Compare output
        expected=$(cat "$expected_file")
        if [ "$actual" = "$expected" ]; then
            echo "[PASS]"
            PASS=$((PASS + 1))
        else
            echo "[FAIL]"
            echo "  Expected: $expected"
            echo "  Actual:   $actual"
            FAIL=$((FAIL + 1))
        fi

        # Cleanup
        rm -f "$output_bin"
    done

    echo ""
    echo "Results: $PASS passed, $FAIL failed, $TOTAL total"

    if [ $FAIL -gt 0 ]; then
        exit 1
    fi
    ```
