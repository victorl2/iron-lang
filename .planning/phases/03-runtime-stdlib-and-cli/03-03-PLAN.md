    **Append to iron_runtime.h** — collection macro definitions:

    ```c
    /* ── Collection macros ──────────────────────────────────────────────── */

    /* List[T] — dynamic array with push, get, set, pop, len, free
     * The codegen emits stub structs like:
     *   typedef struct Iron_List_int64_t { int64_t *items; int64_t count; int64_t capacity; } Iron_List_int64_t;
     * These macros generate the matching function implementations.
     *
     * T = the C type (int64_t, double, bool, Iron_String, Iron_Enemy, etc.)
     * suffix = the exact string used in the mangled name (same as T for primitives)
     */
    #define IRON_LIST_DECL(T, suffix) \
        void Iron_List_##suffix##_push(Iron_List_##suffix *self, T item); \
        T Iron_List_##suffix##_get(const Iron_List_##suffix *self, int64_t index); \
        void Iron_List_##suffix##_set(Iron_List_##suffix *self, int64_t index, T item); \
        T Iron_List_##suffix##_pop(Iron_List_##suffix *self); \
        int64_t Iron_List_##suffix##_len(const Iron_List_##suffix *self); \
        void Iron_List_##suffix##_free(Iron_List_##suffix *self); \
        Iron_List_##suffix Iron_List_##suffix##_create(void);

    #define IRON_LIST_IMPL(T, suffix) \
        Iron_List_##suffix Iron_List_##suffix##_create(void) { \
            Iron_List_##suffix l = {NULL, 0, 0}; return l; \
        } \
        void Iron_List_##suffix##_push(Iron_List_##suffix *self, T item) { \
            if (self->count >= self->capacity) { \
                self->capacity = self->capacity ? self->capacity * 2 : 8; \
                self->items = (T*)realloc(self->items, (size_t)self->capacity * sizeof(T)); \
            } \
            self->items[self->count++] = item; \
        } \
        T Iron_List_##suffix##_get(const Iron_List_##suffix *self, int64_t index) { \
            return self->items[index]; \
        } \
        void Iron_List_##suffix##_set(Iron_List_##suffix *self, int64_t index, T item) { \
            self->items[index] = item; \
        } \
        T Iron_List_##suffix##_pop(Iron_List_##suffix *self) { \
            return self->items[--self->count]; \
        } \
        int64_t Iron_List_##suffix##_len(const Iron_List_##suffix *self) { \
            return self->count; \
        } \
        void Iron_List_##suffix##_free(Iron_List_##suffix *self) { \
            free(self->items); self->items = NULL; self->count = 0; self->capacity = 0; \
        }
    ```

    Similarly define `IRON_MAP_DECL/IRON_MAP_IMPL(K, V, ksuffix, vsuffix)`:
    - Map struct layout: Use stb_ds internally. But the codegen emits a simpler struct. Solution: the Map runtime stores an stb_ds hash map pointer inside the struct but the codegen struct must match.
    - Alternative simpler approach: Map uses an array of key-value pairs with linear search for small maps, and an open-addressing hash table for larger maps. For v1, use simple array-based approach:
      - Struct fields expected by codegen: `<K> *keys; <V> *values; int64_t count; int64_t capacity;`
      - Methods: `put(self, key, value)`, `get(self, key)` (returns value), `has(self, key)` (returns bool), `remove(self, key)`, `len(self)`, `free(self)`
      - put: linear scan for existing key (equality check), if found update value; else append
      - get: linear scan, return value (undefined behavior if not found — caller should check has() first)
      - For v1, O(n) lookup is acceptable; future optimization with hash table can come later

    Define `IRON_SET_DECL/IRON_SET_IMPL(T, suffix)`:
    - Struct fields: `T *items; int64_t count; int64_t capacity;`
    - Methods: `add(self, item)` (no duplicates), `contains(self, item)` (linear scan, returns bool), `remove(self, item)`, `len(self)`, `free(self)`
    - For v1, O(n) containment check is acceptable

    ```c
    #include <stdlib.h>
    #include <string.h>

    /* Pre-instantiate List for common types */
    /* Note: the codegen's ensure_monomorphized_type() emits the struct typedefs.
     * The runtime only needs to provide the function bodies.
     * However, for the runtime's own test file, we need the struct typedefs too.
     * Define them here conditionally (codegen output will have its own). */

    #ifndef IRON_CODEGEN_PROVIDES_STRUCTS
    typedef struct Iron_List_int64_t { int64_t *items; int64_t count; int64_t capacity; } Iron_List_int64_t;
    typedef struct Iron_List_double { double *items; int64_t count; int64_t capacity; } Iron_List_double;
    typedef struct Iron_List_bool { bool *items; int64_t count; int64_t capacity; } Iron_List_bool;
    typedef struct Iron_List_Iron_String { Iron_String *items; int64_t count; int64_t capacity; } Iron_List_Iron_String;

    typedef struct Iron_Map_Iron_String_int64_t { Iron_String *keys; int64_t *values; int64_t count; int64_t capacity; } Iron_Map_Iron_String_int64_t;
    typedef struct Iron_Map_Iron_String_Iron_String { Iron_String *keys; Iron_String *values; int64_t count; int64_t capacity; } Iron_Map_Iron_String_Iron_String;

    typedef struct Iron_Set_int64_t { int64_t *items; int64_t count; int64_t capacity; } Iron_Set_int64_t;
    typedef struct Iron_Set_Iron_String { Iron_String *items; int64_t count; int64_t capacity; } Iron_Set_Iron_String;
    #endif

    /* Generate implementations for common types */
    IRON_LIST_IMPL(int64_t, int64_t)
    IRON_LIST_IMPL(double, double)
    IRON_LIST_IMPL(bool, bool)
    IRON_LIST_IMPL(Iron_String, Iron_String)

    /* Map needs special equality comparison for keys.
     * For Iron_String keys: use iron_string_equals
     * For integer keys: use == */
    IRON_MAP_IMPL(Iron_String, int64_t, Iron_String, int64_t, iron_string_equals_ptr)
    IRON_MAP_IMPL(Iron_String, Iron_String, Iron_String, Iron_String, iron_string_equals_ptr)

    IRON_SET_IMPL(int64_t, int64_t, int64_eq)
    IRON_SET_IMPL(Iron_String, Iron_String, iron_string_equals_ptr)
    ```

    The Map and Set macros need an equality function parameter for key/element comparison. Define helper:
    ```c
    static inline bool iron_string_equals_ptr(const Iron_String *a, const Iron_String *b) {
        return iron_string_equals(a, b);
    }
    static inline bool int64_eq(const int64_t *a, const int64_t *b) {
        return *a == *b;
    }
    ```

    Adjust the MAP and SET macro signatures to include an `eq_fn` parameter for key comparison.
