```c
 * asserts below lock this layout so that `Iron_ExprNode *` reads correctly
```

```c
/* ── Internal compiler error (ICE) helper ────────────────────────────────────
 * iron_ice is the canonical abort path for compiler-internal invariants that
 * should never be reachable in a correct build. It prints a formatted message
 * prefixed with "iron: internal compiler error: " to stderr and calls abort().
 *
 * Use iron_ice when:
 *   - An AST node is the wrong kind after a sym_kind check that should have
 *     guaranteed the correct kind (PROT-03 kind-assert failure path).
 *   - A switch over an Iron_*Kind hits a case the compiler thought was
 *     unreachable.
 *   - An invariant that the type system should have enforced is violated at
 *     runtime.
 *
 * Do NOT use iron_ice for user-facing errors — those go through iron_diag_emit
 * with the Iron_DiagList surface. iron_ice is compiler-bug territory only.
 *
 * The function is declared noreturn. It takes a printf-style format string.
 */
#if defined(__GNUC__) || defined(__clang__)
__attribute__((noreturn, format(printf, 1, 2)))
#endif
void iron_ice(const char *fmt, ...);
```

**Step 2: Add `iron_ice` definition + `iron_node_assert_kind_impl` definition to src/diagnostics/diagnostics.c.**

Append to the end of diagnostics.c. Use `#include &lt;stdarg.h&gt;` and `#include &lt;stdlib.h&gt;` if not already present (grep first to confirm; add to top of file if missing):

```c
/* ── Internal compiler error (PROT-03) ───────────────────────────────────── */

#include <stdarg.h>
#include <stdio.h>
#include <stdlib.h>

void iron_ice(const char *fmt, ...) {
    fprintf(stderr, "iron: internal compiler error: ");
    va_list args;
    va_start(args, fmt);
    vfprintf(stderr, fmt, args);
    va_end(args);
    fprintf(stderr, "\n");
    fflush(stderr);
    abort();
}

/* ── AST node kind assertion impl (PROT-03) ──────────────────────────────── */

/* Forward declared in src/parser/ast.h. Lives here (not in ast.h) so the impl
 * can call iron_ice without pulling the diagnostics surface into ast.h as a
 * transitive dependency. ast.h already includes diagnostics.h for Iron_Span,
 * so the declaration of iron_node_assert_kind_impl piggybacks on that include. */
#include "parser/ast.h"

void iron_node_assert_kind_impl(const Iron_Node *node,
                                Iron_NodeKind expected,
                                const char *file,
                                int line,
                                const char *func) {
    if (!node) {
        iron_ice("iron_node_assert_kind: NULL node (expected kind %d) at %s:%d in %s",
                 (int)expected, file, line, func);
    }
    if (node->kind != expected) {
        iron_ice("iron_node_assert_kind: expected kind %d, got %d at %s:%d in %s",
                 (int)expected, (int)node->kind, file, line, func);
    }
}
```

Note: The message uses integer kind values rather than stringified names because mapping `Iron_NodeKind` to strings would either require a new helper or duplicate the existing printer.c mapping. Integer values are sufficient for bisection and keep the dependency graph simple. If a future phase wants named output, it can add an `iron_node_kind_name(Iron_NodeKind)` helper.

```c
 * Debug-build runtime check that an AST node has the expected kind before a
 * concrete cast. Call immediately before casting Iron_Node* to any concrete
 * expression/statement/decl type:

void iron_node_assert_kind_impl(const Iron_Node *node,
```
